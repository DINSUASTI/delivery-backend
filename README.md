1. SCRIPTS BASE DE DATOS (Postgress)

    - Creacion BD
  
    - -- Database: delivery

           DROP DATABASE IF EXISTS delivery;
          
          CREATE DATABASE delivery
              WITH
              OWNER = postgres
              ENCODING = 'UTF8'
              LC_COLLATE = 'es_ES.UTF-8'
              LC_CTYPE = 'es_ES.UTF-8'
              LOCALE_PROVIDER = 'libc'
              TABLESPACE = pg_default
              CONNECTION LIMIT = -1
              IS_TEMPLATE = False;

-- Table: public.webhook_log

        DROP TABLE IF EXISTS public.webhook_log;

        CREATE TABLE IF NOT EXISTS public.webhook_log
        (
            id bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1 ),
            action character varying(255) COLLATE pg_catalog."default" NOT NULL,
            event_type character varying(255) COLLATE pg_catalog."default" NOT NULL,
            message character varying(1024) COLLATE pg_catalog."default",
            payload_summary character varying(2048) COLLATE pg_catalog."default",
            response_id bigint,
            source character varying(255) COLLATE pg_catalog."default" NOT NULL,
            status_code integer,
            "timestamp" timestamp(6) with time zone NOT NULL,
            CONSTRAINT webhook_log_pkey PRIMARY KEY (id)
        )

      TABLESPACE pg_default;
      
      ALTER TABLE IF EXISTS public.webhook_log
          OWNER to postgres;


-- Table: public.webhook_response

        DROP TABLE IF EXISTS public.webhook_response;
        
        CREATE TABLE IF NOT EXISTS public.webhook_response
        (
            id bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1 ),
            event_type character varying(255) COLLATE pg_catalog."default" NOT NULL,
            file_content bytea,
            file_name character varying(255) COLLATE pg_catalog."default" NOT NULL,
            file_type character varying(255) COLLATE pg_catalog."default" NOT NULL,
            message character varying(512) COLLATE pg_catalog."default",
            received_at timestamp(6) with time zone NOT NULL,
            source character varying(255) COLLATE pg_catalog."default" NOT NULL,
            state character varying(255) COLLATE pg_catalog."default" NOT NULL,
            idempotency_key character varying(64) COLLATE pg_catalog."default",
            CONSTRAINT webhook_response_pkey PRIMARY KEY (id),
            CONSTRAINT uknwvfi7i8wt5ivik0sh46mld1x UNIQUE (idempotency_key),
            CONSTRAINT webhook_response_state_check CHECK (state::text = ANY (ARRAY['PENDING'::character varying, 'EXTERNAL_CALL_IN_PROGRESS'::character varying, 'EXTERNAL_CALL_RETRY'::character varying, 'PROCESSING'::character varying, 'SUCCESS'::character varying, 'FAILED_TRANSIENT'::character varying, 'FAILED_PERMANENT'::character varying, 'FAILED'::character varying]::text[]))
        )
        
        TABLESPACE pg_default;
        
        ALTER TABLE IF EXISTS public.webhook_response
            OWNER to postgres;
